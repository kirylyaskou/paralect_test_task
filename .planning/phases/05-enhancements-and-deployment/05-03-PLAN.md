      ```typescript
      async function uploadFiles(attachments: Attachment[], chatId: string): Promise<{
        imageUrls: string[]
        documentUrls: Array<{ fileName: string; fileUrl: string; extractedText?: string }>
      }> {
        const imageUrls: string[] = []
        const documentUrls: Array<{ fileName: string; fileUrl: string; extractedText?: string }> = []

        for (const att of attachments) {
          // Step 1: Get signed URL
          const res = await fetch('/api/upload', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              filename: att.file.name,
              contentType: att.file.type,
              size: att.file.size,
              chatId,
            }),
          })
          if (!res.ok) {
            const err = await res.json()
            throw new Error(err.error || 'Failed to get upload URL')
          }
          const { signedUrl, token, path } = await res.json()

          // Step 2: Upload directly to Supabase Storage
          const { error } = await supabasePublic.storage
            .from('attachments')
            .uploadToSignedUrl(path, token, att.file, {
              contentType: att.file.type,
            })
          if (error) throw new Error('Failed to upload file')

          // Step 3: Complete upload (extraction for docs)
          const completeRes = await fetch('/api/upload/complete', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
              path,
              chatId,
              fileName: att.file.name,
              fileType: att.file.type,
            }),
          })
          if (!completeRes.ok) throw new Error('Failed to process upload')
          const result = await completeRes.json()

          if (att.type === 'image') {
            imageUrls.push(result.fileUrl)
          } else {
            documentUrls.push({
              fileName: att.file.name,
              fileUrl: result.fileUrl,
              extractedText: result.extractedText,
            })
          }
        }
        return { imageUrls, documentUrls }
      }
      ```
